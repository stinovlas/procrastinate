36def load_from_path(path: str, allowed_type: type[T] | None = None) -> T: 

37 """ 

38 Import and return then object at the given full python path. 

39 """ 

40 if "." not in path: 

41 raise exceptions.LoadFromPathError(f"{path} is not a valid path") 

42 module_path, name = path.rsplit(".", 1) 

43 try: 

44 module = importlib.import_module(module_path) 

45 except ImportError as exc: 

46 raise exceptions.LoadFromPathError(str(exc)) from exc 

47 

48 try: 

49 imported = getattr(module, name) 

50 except AttributeError as exc: 

51 raise exceptions.LoadFromPathError(str(exc)) from exc 

52 

53 if allowed_type and not isinstance(imported, allowed_type): 

54 raise exceptions.LoadFromPathError( 

55 f"Object at {path} is not of type {allowed_type.__name__} " 

56 f"but {type(imported).__name__}" 

57 ) 

58 

59 return imported

74def caller_module_name(prefix: str = "procrastinate") -> str: 

75 """ 

76 Returns the module name of the first module of the stack that isn't under ``prefix``. 

77 If any problem occurs, raise `CallerModuleUnknown`. 

78 """ 

79 

80 try: 

81 frame = inspect.currentframe() 

82 while True: 

83 if not frame: 

84 raise ValueError("Empty frame") 

85 name = frame.f_globals["__name__"] # May raise ValueError 

86 if not name.startswith(f"{prefix}."): 

87 break 

88 frame = frame.f_back 

89 return name 

90 except Exception as exc: 

91 raise exceptions.CallerModuleUnknown from exc

122def _get_module_name(obj: Any) -> str: 

123 module_name = obj.__module__ 

124 if module_name != "__main__": 

125 return module_name 

126 

127 module = inspect.getmodule(obj) 

128 # obj could be None, or has no __file__ if in an interactive shell 

129 # in which case, there's not a lot we can do. 

130 if not module or not getattr(module, "__file__", None): 

131 return module_name 

132 

133 # We've checked this just above but mypy needs a direct proof 

134 assert module.__file__ 

135 

136 path = pathlib.Path(module.__file__) 

137 # If the path is absolute, it probably means __main__ is an executable from an 

138 # installed package 

139 if path.is_absolute(): 

140 return module_name 

141 

142 # Creating the dotted path from the path 

143 return ".".join([*path.parts[:-1], path.stem])

160def parse_datetime(raw: str) -> datetime.datetime: 

161 try: 

162 # this parser is the stricter one, so we try it first 

163 dt = dateutil.parser.isoparse(raw) 

164 if not dt.tzinfo: 

165 dt = dt.replace(tzinfo=datetime.timezone.utc) 

166 return dt 

167 except ValueError: 

168 pass 

169 # this parser is quite forgiving, and will attempt to return 

170 # a value in most circumstances, so we use it as last option 

171 dt = dateutil.parser.parse(raw) 

172 dt = dt.replace(tzinfo=datetime.timezone.utc) 

173 return dt

220async def run_tasks( 

221 main_coros: Iterable[Coroutine], 

222 side_coros: Iterable[Coroutine] | None = None, 

223 graceful_stop_callback: Callable[[], Any] | None = None, 

224): 

225 """ 

226 Run multiple coroutines in parallel: the main coroutines and the side 

227 coroutines. Side coroutines are expected to run until they get cancelled. 

228 Main corountines are expected to return at some point. By default, this 

229 function will return None, but on certain circumstances, (see below) it can 

230 raise a `RunTaskError`. A callback `graceful_stop_callback` will be called 

231 if provided to ask the main coroutines to gracefully stop in case either 

232 one of them or one of the side coroutines raise. 

233 

234 - If all coroutines from main_coros return and there is no exception in the 

235 coroutines from either `main_coros` or `side_coros`: 

236 - coroutines from `side_coros` are cancelled and awaited 

237 - the function return None 

238 

239 - If any corountine from `main_coros` or `side_coros` raises an exception: 

240 - `graceful_stop_callback` is called (the idea is that it should ask 

241 coroutines from `main_coros` to exit gracefully) 

242 - the function then wait for main_coros to finish, registering any 

243 additional exception 

244 - coroutines from `side_coros` are cancelled and awaited, registering any 

245 additional exception 

246 - all exceptions from coroutines in both `main_coros` and `side_coros` 

247 are logged 

248 - the function raises `RunTaskError` 

249 

250 It's not expected that coroutines from `side_coros` return. If this 

251 happens, the function will not react in a specific way. 

252 

253 When a `RunTaskError` is raised because of one or more underlying 

254 exceptions, one exception is the `__cause__` (the first main or side 

255 coroutine that fails in the input iterables order, which will probably not 

256 the chronologically the first one to be raised). All exceptions are logged. 

257 """ 

258 # Ensure all passed coros are futures (in our case, Tasks). This means that 

259 # all the coroutines start executing now. 

260 # `name` argument to create_task only exist on python 3.8+ 

261 main_tasks = [asyncio.create_task(coro, name=coro.__name__) for coro in main_coros] 

262 side_tasks = [ 

263 asyncio.create_task(coro, name=coro.__name__) for coro in side_coros or [] 

264 ] 

265 for task in main_tasks + side_tasks: 

266 name = task.get_name() 

267 logger.debug( 

268 f"Started {name}", 

269 extra={ 

270 "action": f"{name}_start", 

271 }, 

272 ) 

273 

274 # Note that asyncio.gather() has 2 modes of execution: 

275 # - asyncio.gather(*aws) 

276 # Interrupts the gather at the first exception, and raises this 

277 # exception. Otherwise, return a list containing return values for all 

278 # coroutines 

279 # - asyncio.gather(*aws, return_exceptions=True) 

280 # Run every corouting until the end, return a list of either return 

281 # values or raised exceptions (mixed). 

282 

283 # The _main function will always raise: either an exception if one happens 

284 # in the main tasks, or EndMain if every coroutine returned 

285 async def _main(): 

286 await asyncio.gather(*main_tasks) 

287 raise EndMain 

288 

289 exception_records: list[ExceptionRecord] = [] 

290 try: 

291 # side_tasks supposedly never finish, and _main always raises. 

292 # Consequently, it's theoretically impossible to leave this try block 

293 # without going through one of the except branches. 

294 await asyncio.gather(_main(), *side_tasks) 

295 except EndMain: 

296 pass 

297 except Exception as exc: 

298 logger.error( 

299 "Main coroutine error, initiating remaining coroutines stop. " 

300 f"Cause: {exc!r}", 

301 extra={ 

302 "action": "run_tasks_stop_requested", 

303 }, 

304 ) 

305 if graceful_stop_callback: 

306 graceful_stop_callback() 

307 

308 # Even if we asked the main tasks to stop, we still need to wait for 

309 # them to actually stop. This may take some time. At this point, any 

310 # additional exception will be registered but will not impact execution 

311 # flow. 

312 results = await asyncio.gather(*main_tasks, return_exceptions=True) 

313 for task, result in zip(main_tasks, results): 

314 if isinstance(result, Exception): 

315 exception_records.append( 

316 ExceptionRecord( 

317 task=task, 

318 exc=result, 

319 ) 

320 ) 

321 else: 

322 name = task.get_name() 

323 logger.debug( 

324 f"{name} finished execution", 

325 extra={ 

326 "action": f"{name}_stop", 

327 }, 

328 ) 

329 

330 for task in side_tasks: 

331 task.cancel() 

332 try: 

333 # task.cancel() says that the next time a task is executed, it will 

334 # raise, but we need to give control back to the task for it to 

335 # actually recieve the exception. 

336 await task 

337 except asyncio.CancelledError: 

338 name = task.get_name() 

339 logger.debug( 

340 f"Stopped {name}", 

341 extra={ 

342 "action": f"{name}_stop", 

343 }, 

344 ) 

345 except Exception as exc: 

346 exception_records.append( 

347 ExceptionRecord( 

348 task=task, 

349 exc=exc, 

350 ) 

351 ) 

352 

353 for exception_record in exception_records: 

354 name = exception_record.task.get_name() 

355 message = f"{name} error: {exception_record.exc!r}" 

356 action = f"{name}_error" 

357 logger.exception( 

358 message, 

359 extra={ 

360 "action": action, 

361 }, 

362 ) 

363 

364 if exception_records: 

365 raise exceptions.RunTaskError from exception_records[0].exc
